type Links struct {
	Level int
	Link  string
}

var tokens = make(chan struct{}, 10)

func main() {
	linkList := make(chan Links)
	go func() {
		for _, v := range os.Args[1:] {
			linkList <- Links{1, v}
		}
	}()
	depth := 4
	m := make(map[string]bool)
	for link := range linkList {
		if !m[link.Link] && link.Level < depth {
			m[link.Link] = true
			tokens <- struct{}{}
			go func(link Links) {
				fmt.Printf("link = %+v\n", link)
				res := utility.Parser(link.Link)
				<-tokens
				for _, v := range res {
					linkList <- Links{link.Level + 1, v}
				}
			}(link)
		}
	}
}

utility.go

func Parser(url string) []string {
	resp, err := http.Get(url)
	if err != nil {
		fmt.Errorf("http get %s with error %v", url, err)
		return nil
	}
	defer resp.Body.Close()
	doc, err := html.Parse(resp.Body)
	if err != nil {
		fmt.Errorf("parse problem %s with error %v", url, err)
		return nil
	}
	c := &Client{url: resp.Request.URL}
	pre := c.findLink
	res := forEachNode(doc, pre, nil)
	return res
}

type Client struct {
	url *url.URL
}

func (c *Client) findLink(n *html.Node) (res []string) {
	if n.Type == html.ElementNode && n.Data == "a" {
		for _, v := range n.Attr {
			if v.Key != "href" {
				continue
			}
			temp, err := c.url.Parse(v.Val)
			if err != nil {
				continue
			}
			res = append(res, temp.String())
		}
	}
	return
}
func forEachNode(n *html.Node, pre, post func(n *html.Node) []string) (res []string) {
	if pre != nil {
		res = append(res, pre(n)...)
	}
	for c := n.FirstChild; c != nil; c = c.NextSibling {
		res = append(res, forEachNode(c, pre, post)...)
	}
	if post != nil {
		res = append(res, post(n)...)
	}
	return
}
